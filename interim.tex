\section{YANG-CBOR 'instance-identifier' draft}
\begin{frame}[fragile]
	\frametitle{YANG-CBOR 'instance-identifier' draft}
	YANG built-in type 'instance-identifier' is type for indirect references.
	It must point to a specific instance node in data tree.

\begin{verbatim}
// file ex.yang
container system {
  container services {
    container ssh {
      leaf port { type uint16; }
    }
  }
  list server {
    key "ip port";
    leaf ip { type inet:ip-address; }
    leaf port { type uint16; }
    leaf service { type string; }
  }
}
\end{verbatim}
\end{frame}

\begin{frame}[fragile]

	\frametitle{YANG-CBOR 'instance-identifier' draft}
\begin{verbatim}
/* instance-identifier for a container */
N: /ex:system/ex:services/ex:ssh (SID 1200)
S: 1200

/* instance-identifier for a leaf */
N: /ex:system/ex:services/ex:ssh/ex:port (SID 1201)
S: 1201

/* instance-identifier for a list entry */
N: /ex:system/ex:server[ex:ip='192.0.0.1'][ex:port='80']\
  (SID 1230)
S: [1230, "192.0.0.1", 80]

/* instance-identifier for a leaf in a list entry */
N: /ex:system/ex:user[ex:ip='192.0.0.1'][ex:port='80']\
  /ex:service (SID 1237)
S: [1237, "192.0.0.1", 80]
\end{verbatim}
\end{frame}

\begin{frame}[fragile]
	\frametitle{YANG-CBOR 'instance-identifier' draft}
	Not covered cases
\begin{verbatim}
// file ex2.yang
container system {
  list logged-users {
    config false;
    // no key stmt;
    leaf name { type string; }
    leaf number-of-sessions { type uint32; }
  }
  leaf-list if-names { type string; }
  leaf-list measurements { type uint64; }
}
\end{verbatim}
\end{frame}

\begin{frame}[fragile]
	\frametitle{YANG-CBOR 'instance-identifier' draft}
Proposed YANG-CBOR encoding using SIDs:

\begin{verbatim}
/* instance-identifier for a list entry with two keys */
N: /ex2:system/ex2:logged-users[2] (SID 1241)
draft S: [1241, 2]

/* instance-identifier for a leaf-list entry */
N: /ex2:system/ex2:if-names[.='eth0'] (SID 1260)
draft S: [1260, "eth0"]

N: /ex2:system/ex2:measurements[.='1440'] (SID 1261)
draft S: [1261, 1440]
\end{verbatim}

For more details see draft text.
\end{frame}

\section{YANG Stand-ins}
\begin{frame}[fragile]
	\frametitle{YANG Stand-ins}
	
	Most of YANG types are based on string built-in type.
	This is not ideal for constrained environments.
	We create an optimalization for types that have more natural and/or efficient 
	representation in CBOR.
	
	Examples:
	
	\verb|ip4-address|, \verb|ip6-address|, \verb|uuid|, \verb|mac-address|, \verb|date|, \verb|date-and-time|.

\end{frame}

\begin{frame}[fragile]
	\frametitle{YANG Stand-ins}
	
	\begin{block}{Standin file}
	\begin{verbatim}
module ietf-cbor-standin-file { container standin-file {
  list encoding {
  config false;
    leaf standin { type identityref; }
    choice encoding-variant {
      list types { key "type"; leaf type { type string; }}
      list sids { key "sid"; leaf sid { type sid:sid; }}
    }
  }
} }
	\end{verbatim}
	\end{block}

\end{frame}

\begin{frame}[fragile]
	\frametitle{YANG Stand-ins}
	
	PR: \#15
	
	Encode simple cases of \verb|enumeration| and \verb|bits| inside \verb|union|
	as CBOR unsigned integers.
	Only if the values are unique.
\begin{verbatim}
leaf unioned {
  type union {
    type string;
    type enumeration { enum first; enum second; enum third; }
    type bits { bit b1; bit b2; bit b3; }  
  }
}
\end{verbatim}

\begin{verbatim}
OLD                 NEW
"string-value"      "string-value"
"first"             44(0)
"b1 b3"             43(b"3")
\end{verbatim}
		
\end{frame}
